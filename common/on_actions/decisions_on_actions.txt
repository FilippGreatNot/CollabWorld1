on_actions = {
	on_startup = {
		effect = {
			every_country = {
				set_variable = { decision_generic_agitation_cost = modifier@decision_generic_agitation_cost }
				set_variable = { decision_generic_war_propaganda_cost = modifier@decision_generic_war_propaganda_cost }
			} 
		}
	}
	on_daily = {
		effect = {
			every_country = {
				set_variable = { decision_generic_agitation_cost = modifier@decision_generic_agitation_cost } #The game updates the cost only every month if modifier is set directly
				set_variable = { decision_generic_war_propaganda_cost = modifier@decision_generic_war_propaganda_cost }
			}
		}
	}
}
