on_actions = {
	on_capitulation = { ##making sure that once dzungaria capitulates, its actually annexed
        effect = { 
	        if = { 
				limit = {
					ROOT = {
						original_tag = DZG
					}
					FROM = {
						original_tag = ALT
					}
				}
				ALT = {
					annex_country = {
						target = DZG
						transfer_troops = no
					}
				}
			}
        }
	}
	on_capitulation = { 
        effect = { 
	        if = { 
				limit = {
					ROOT = {
						original_tag = RMA
					}
					FROM = {
						original_tag = ALT
						has_country_flag = ALT_puppets_rma_true
					}
				}
				RUS = {
					end_puppet = RMA
				}
				ALT = {
				    white_peace = RMA
					white_peace = RUS
					set_autonomy = {
	            		target = RMA
	            		autonomous_state = autonomy_puppet
                	}
					country_event = alt.2
				}
				RMA = {
					set_cosmetic_tag = RMA_ALT
					MON = { make_every_core_national_for_PREV = yes }
					set_capital = { state = 1006 remember_old_capital = no }
				}
				818 = { 
					add_core_of = ALT
					transfer_state_to = ALT
				}
				329 = {
					add_core_of = RMA
					add_core_of = MON
					remove_core_of = RUS
					transfer_state_to = RMA
				}
			}
		}
	}
}
