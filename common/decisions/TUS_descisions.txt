TUS_jump_itas = {
	TUS_jump_its = {
		available = {
			is_neighbor_of = ITS
			ITS = {has_war = yes}
			NOT = {has_war_with = ROOT}
			country_exists = ITS
		}
		visible = {country_exists = ITS}
		
		days_remove = 10
		fire_only_once = yes

		remove_effect	= {
			if = {
				limit = {country_exists = ITS}
				declare_war_on = {
					target = ITS
					type = annex_everything
				}
				add_timed_idea = {
					idea = TUS_unplanned_attack
					days = 30
				}
			}
		}
		ai_will_do = {
			factor = 0
		}
	}
