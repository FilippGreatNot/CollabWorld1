country_event = {
	id = tus_coun1.1
	title = tus_coun1.1.t
	desc = tus_coun1.1.d
	
	fire_only_once = yes
	
	trigger = {
		focus_progress = {
			focus = TUS_unify_the_party
			progress > 0
		}	
	}
	
	option = {
		name = tus_coun1.1.a
		country_event = {
			id = tus_coun1.2
			days = 35
		}
		add_to_variable = {
			var = TUS_Balbo_Pop
			value = 1
		}
	}
	option = {
		name = tus_coun1.1.b
		add_ideas = a_gun_for_every_household
		id = tus_coun1.2
		days = 35
	}
}

country_event = {
	id = tus_coun1.2
	title = tus_coun1.2.t
	desc = tus_coun1.2.d
	is_triggered_only = yes
	
	option = {
		name = tus.5.a
		add_popularity = {
			ideology = national_syndicalism
			popularity = 0.1
		}
		if = {
			limit = {check_variable = {TUS_Balbo_Pop > 6}}
			country_event = {
				id = tus_coun1.3
				days = 35
			}
		}
		else = {
			country_event = {
				id = tus_coun1.4
				days = 35
			}
		}
	}
	option = {
		name = tus.5.b

		add_popularity = {
			ideology = national_syndicalism

			limit = {check_variable = {TUS_Balbo_Pop > 6}}
			country_event = {
				id = tus_coun1.3
				days = 35
			}
		}
		else = {
			country_event = {
				id = tus_coun1.4
				days = 35
			}
		}
	}
}
