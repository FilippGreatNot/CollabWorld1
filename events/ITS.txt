## Main ##

country_event = {
	id = its.1
	title = its.1.t
	desc = its.1.d
	
	fire_only_once = yes
	
	trigger = {
		focus_progress = {
			focus = ITS_assure_ppl
			progress > 0.5
		}	
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = ITS_assassination_attempt
			unlock_national_focus = ITS_assure_ppl
			mark_focus_tree_layout_dirty = yes
		}
	}
	
	option = {
		name = its.1.a
		complete_national_focus = ITS_he_lives
	}
	option = {
		name = its.1.b
		complete_national_focus = ITS_he_died
	}
}

## Coverup ##

country_event = {
	id = its_coverup.1
	title = its_coverup.1.t
	desc = its_coverup.1.d
	fire_only_once = yes
	
	trigger = {
		focus_progress = {
			focus = ITS_investigate
			progress > 0
		}	
	}
	
	immediate = {
		hidden_effect = {
			country_event = {
				id = its_coverup.2
				days = 20
			}			
		}
		set_variable = {
			var = ITS_coverup_points
			value = 0
		}
	}
	
	option = {
		#kill assassin
		name = its_coverup.1.a
	}
	option = {
		#let the assassin live
		name = its_coverup.1.b
		add_to_variable = {
			var = ITS_coverup_points
			value = 1
		}
	}
}
