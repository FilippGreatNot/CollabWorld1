country_event = {
	id = its.1
	title = its.1.t
	desc = its.1.d
	
	fire_only_once = yes
	
	trigger = {
		focus_progress = {
			focus = ITS_assure_ppl
			progress > 0.5
		}	
		check_variable = {
			kill_mus_points > 80
		}
	}

		hidden_effect = {
			set_country_flag = ITS_assassination_attempt

			mark_focus_tree_layout_dirty = yes
			kill_country_leader = yes
			promote_character = ITS_emergency_council		
		}
	}

	option = {
		name = its.1.a
		add_stability = -0.1
	}
}
