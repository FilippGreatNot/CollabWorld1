country_event = {
	id = sov_unification.1 #tukh
	immediate = { log = "[GetDateText]: [Root.GetName]: event powerstruggle.1" }
	title = sov_unification.1.t
	desc = sov_unification.1.d
	picture = GFX_report_event_soviet_people
	is_triggered_only = yes
	
	option = {
		name = sov_unification.1.a #####Тухач Согласился
		ai_chance = { factor = 90 }
		SOV = {
			annex_country = {
				target = PBD
				transfer_troops = yes 
			}
		}
		
		PBD = {
			every_unit_leader = {
				set_nationality = SOV
			}
			if = {
				limit = {
					SOV = {
						has_country_leader = { character = SOV_joseph_stalin ruling_only = yes }
					}
				}
		
				SOV = {
					complete_national_focus = SOV_peaceful_integration		
				}
			}
			
			if = {
				limit = {
					SOV = {
						has_country_leader = { character = SOV_alexander_bogdanov ruling_only = yes }
					}
				}
				
				SOV = {
					complete_national_focus = SOV_welcome_home_tukh
				} 
			}
		}
	}
	
	option = {
		name = sov_unification.1.b	
		ai_chance = { factor = 10 }
		SOV = {
			declare_war_on = {
				target = PBD
				type = annex_everything
			}
		}	
		
		if = {
			limit = {
				SOV = {
					has_country_leader = { character = SOV_joseph_stalin ruling_only = yes }
				}
			}
			
			SOV = {
				complete_national_focus = SOV_teach_the_impudent_marshal_a_lesson
			} 
		}
	
		if = {
			limit = {
				SOV = {
					has_country_leader = { character = SOV_alexander_bogdanov ruling_only = yes }
				}
			}
			
			
			SOV = {
				complete_national_focus = SOV_this_is_treason_marshal
			}			
		}
	}
}
